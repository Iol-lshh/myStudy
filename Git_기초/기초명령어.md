git init   
    
git clone {url}   
   
git add {파일명}    
: 스테이징 상태로 올림   
   
   
git status -s   
    
git diff   
git diff --cached    
git diff --statged   
: Staged 된 파일들을 확인   
   
git commit -m "{message}"   
: 커밋   
-v option : add diff contents   
   
git config --global user.email {이메일}
: 이메일 입력
git config user.email
: 이메일 확인
   
git config --global user.name {유저네임}
: 사용자 이름 입력
git config user.name
: 사용자 이름 확인
   
git remote -v
: 리모트 위치 확인
   
git fetch {리모트 이름}
: 가져오기
   
git push {리모트 저장소 이름} {브랜지 이름}
: 원격에 넣기
   
git remote show {리모트 저장소 이름}
: 리모트 저장소 상태 확인
